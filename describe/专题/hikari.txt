private boolean shouldCreateAnotherConnection() {
	return getTotalConnections() < config.getMaximumPoolSize() &&
        (connectionBag.getWaitingThreadCount() > 0 || getIdleConnections() < config.getMinimumIdle());
}



总连接数 < 配置最大连接数
等待线程数 > 0 || 空闲连接 < 配置最小空闲数
